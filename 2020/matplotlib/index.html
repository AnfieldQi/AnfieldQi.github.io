bar柱状图

import matplotlib.pyplot as plt
import numpy as np

n = 12
X = np.arange(n)
Y1 = (1 - X / float(n)) * np.random.uniform(0.5, 1.0, n)
Y2 = (1 - X / float(n)) * np.random.uniform(0.5, 1.0, n)

plt.bar(X, +Y1, facecolor='#9999ff', edgecolor='white')
plt.bar(X, -Y2, )

#为柱状图设置标签
#ha:水平对齐方式
#va:竖直对齐方式
for x, y in zip(X, Y1):
    plt.text(x + 0.04, y + 0.05, '%.2f' % y, ha='center', va='bottom')

for x, y in zip(X, Y2):
    plt.text(x + 0.04, -y-0.05, '%.2f' % y, ha='center', va='top')

plt.xlim(-0.5, n)
plt.xticks(())
plt.ylim(-1.25, 1.25)
plt.yticks(())

plt.show()

Contours等高线

import matplotlib.pyplot as plt
import numpy as np

import matplotlib.pyplot as plt
import numpy as np

def f(x,y):
    # the height function
    return (1 - x / 2 + x**5 + y**3) * np.exp(-x**2 -y**2)

#生成数据做为底图网格
n = 256
x = np.linspace(-3, 3, n)
y = np.linspace(-3, 3, n)
X,Y = np.meshgrid(x, y) #底图网格

#contourf颜色填充网格
#cmap：生成的数据对应得颜色,填充
#8：等高线分为多少段
plt.contourf(X,Y,f(X,Y),8,alpha=0.75,cmap=plt.cm.hot)

#画等高线的线
C=plt.contour(X,Y,f(X,Y),8,colors='black',linewidth=0.5)

#加入等高线高度值
#线画给C，画在线里面，大小等于10
plt.clabel(C,inline=True,fontsize=10)

plt.xticks(())
plt.yticks(())

plt.show()

image 图片

import matplotlib.pyplot as plt
import numpy as np
#随机矩阵画图
a = np.array([0.313660827978, 0.365348418405, 0.423733120134,
              0.365348418405, 0.439599930621, 0.525083754405,
              0.423733120134, 0.525083754405, 0.651536351379]).reshape(3,3)
#imshow:数据图形化
#origin:位置的设定
#interpolation：多种不同的网格效果
plt.imshow(a,interpolation='nearest',cmap=plt.cm.bone,origin='lower')

plt.colorbar(shrink=0.9)

plt.show()

3D数据

import matplotlib.pyplot as plt
import numpy as np
from mpl_toolkits.mplot3d import Axes3D

fig=plt.figure()
ax=Axes3D(fig)
X=np.arange(-4,4,0.25)
Y=np.arange(-4,4,0.25)
X,Y=meshgrid(X,Y)
R=np.sqrt(X**2+Y**2)
#height value
Z=np.sin(R)
#rstride=1,cstride=1:跨度
ax.plot_surface(X,Y,Z,rstride=1,cstride=1,cmap=plt.get_cmap('rainbow'))

#zdir等高线压到哪个值
#offset:压到哪个值面
ax.contourf(X,Y,Z,zdir='z',offset=-2,camp='rainbow')

plt.show()

import matplotlib.pyplot as plt

# 初始化figure
fig = plt.figure()

# 创建数据
x = [1, 2, 3, 4, 5, 6, 7]
y = [1, 3, 4, 2, 5, 8, 6]
left, bottom, width, height = 0.1, 0.1, 0.8, 0.8
ax1=fig.add_axes([left, bottom, width, height])
ax1.plot(x,y,'r')
ax1.set_xlabel('x')
ax1.set_ylabel('y')
ax1.set_title('title')
#方法1
left, bottom, width, height = 0.2, 0.6, 0.25, 0.25
ax2=fig.add_axes([left, bottom, width, height])
ax2.plot(x,y,'r')
ax2.set_xlabel('x')
ax2.set_ylabel('y')
ax2.set_title('title')
#方法2
plt.axes([0.6, 0.2, 0.25, 0.25])
plt.plot(y[::-1], x, 'g') # 注意对y进行了逆序处理
plt.xlabel('x')
plt.ylabel('y')
plt.title('title inside 2')

1.2 Requests

打开网页类型：Get返回/Post：需要发送信息

import requests
import webbrowser
param = {"wd": "Python"}  # 搜索的信息
r = requests.get('http://www.baidu.com/s', params=param)
print(r.url)
webbrowser.open(r.url)
#返回URL不会有个人信息出现
#使用开发者窗口看数据交互
r = requests.post('http://www.baidu.com/s', params=param)

#登录
payload = {'username': 'Morvan', 'password': 'password'}
r = requests.post('http://pythonscraping.com/pages/cookies/welcome.php', data=payload)
print(r.cookies.get_dict())

# {'username': 'Morvan', 'loggedin': '1'}
#方法1：将cookie传递到下一个网页中
r = requests.get('http://pythonscraping.com/pages/cookies/profile.php', cookies=r.cookies)
print(r.text)
#方法2：使用session
session = requests.Session()
payload = {'username': 'Morvan', 'password': 'password'}
r = session.post('http://pythonscraping.com/pages/cookies/welcome.php', data=payload)
print(r.cookies.get_dict())
r = session.get("http://pythonscraping.com/pages/cookies/profile.php")
print(r.text)

1.3 下载文件

import os
from urllib.request import urlretrieve
os.makedirs('./img/', exist_ok=True)

IMAGE_URL = "https://morvanzhou.github.io/static/img/description/learning_step_flowchart.png"
#方法1
urlretrieve(IMAGE_URL, './img/image1.png')
#方法2：完全内存存储后保存
import requests
r = requests.get(IMAGE_URL)
with open('./img/image2.png', 'wb') as f:
    f.write(r.content)
#方法3，边下载边保存
r = requests.get(IMAGE_URL, stream=True)    # stream loading

with open('./img/image3.png', 'wb') as f:
    for chunk in r.iter_content(chunk_size=32):
        f.write(chunk)

#爬取中国地理某栏目图片。就是个匹配选择加控制的过程
import requests
import re
base_url='http://www.dili360.com'
hits=['/Gallery/cate/6/1.htm']
URL=base_url+hits[-1]
html = requests.get(URL).text
soup = BeautifulSoup(html, 'lxml')
img_ul = soup.find_all('div', {"class": "img"})
sub_url=soup.find_all('a',{"href":re.compile("/Gallery*")})
for i in sub_url:
        print(i['href'])
        if i['href'] not in hits:
            hits.append(i['href'])
print(hits)

for _ in range(len(hits)):
    URL=base_url+hits[-1]
    html=requests.get(URL).text
    soup=Beautifulsoup(html,'lxml')
    img_url=soup.find_all('div',{"class":"img"})
    for ul in img_ul:
      imgs=ul.find_all("img")
      for img in imgs:
        url=img['src']
        r=requests.get(url)
        image_name=url.split('/')[-1].split('@')[0]
        with open('./%s'%image_name,'wb') as f:
                  f.write(r.content)

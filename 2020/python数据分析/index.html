Numpy

C语言写的，使用矩阵计算，速度与python相比天差地别

  1. Numpy的属性

    import numpy as np
   
array=np.array([[1,2,3],[4,5,6]])
#array有很多固有属性
array.ndim #维度
array.shape #行列数
array.size #元素个数
array.dtype
np.sum(array,axis=1)#行中求和
np.min(array)
np.max(array)

  2. numpy创建不同array

    import numpy as np
#创建数组
#array没有，分割，与列表不同
a=np.array([2,23,4],dtype=np.int) #声明列表类型dtype:int float常用
a=np.zeros((3,4)) #申请全部为0的矩阵,传入参数
a=np.ones((3,4),dtype=np.int)#申请全0
a=np.empty((3,4))#整个空的
a=np.arange(10,20,2) #申请自增数组
a=np.arange(12).reshape((3,4)) #规定矩阵形状
a=np.linespace(1,10,20)#在范围内生成多少段的数组
a=np.random.random((2,4)) #从0-1随机选择

  3. numpy的基础运算

    import numpy as np
a=np.array([10,20,30,40])
b=np.arange(4)
d=np.array([1,1],[2,2])
e=np.array([1,1],[2,2])
print(a,b)
   
#加减法
c=a-b
print(c)
   
#乘除法
c=b**2
print(c)
   
#三角函数
c=10*np.sin(a)
print(c)
   
#判断数组中的值
print(b<3)#显示一个TF列表，对应元素是否符合表达式
   
#矩阵的运算
c=a*b #元素逐个相乘
c_d=np.dot(a,b) #两个矩阵相乘法则运算
c_2=a.dot(b)
   

    import numpy as np
a=np.arange(2,14).reshape((3,4))
np.argmin(a) #最小值的索引
np.argmax(a)
np.mean(a，axis=) #平均值
np.median(a)#中位数
np.cumsum(a)元素从头到尾逐步相加赋值。即其与前边值的和
np.diff(a)#每两个数之间的差
np.nonzero(a) #输出非0的
np.sort(a) #排序，逐行排序
np.transpose(a) #矩阵转置,用于矩阵乘法
np.clip(a,min,max)#小于min都变为min,大于max变为max,中间不变

  4. numpy索引

    import numpy as np
a=np.arange(3,15)
print(a[3])#支持索引
   
b=np.arange(3,15).reshape((3,4))
print(b[2,:]) #输出一行
   
for row in b: #默认迭代行
    print(row)
   
for column in b.T:#列
    print(column)
#flatten()返回一个被改变的array
#b.flat是一个迭代器，为object
print b.flatten()
for item in b.flat:
    print(item)

#添加空间，将数组变为矩阵
a[:,np.newaxis]
   
#进行多个array的横纵向合并，方法之一，上边那个一样可以

  6. numpy array分割

    import numpy as np
   
a=np.arange(12).reshape((3,4))
print(a)
    
print(np.split(a,2,1))#a:矩阵 2：分成两块 1，按列分（分割行）
   
np.array_split(a,3,1)#不等量分割
   
#横纵向分割
np.vsplit(a,3)
np.hsplit(a,2)

import tkinter as tk
windows=tk.TK()
windows.title()
windows=geometry('num x num ')
#windows上面的label,多参数可选,定义一个变量
var=tk.StringVar()
l=tk.Label(windows,text=var,bg='',font=("Arial",12),width=,height= )
#安置label的方法
l.pack()
on_hit=False
def hit_me():
    global on_hit 
    if on_hit==False:
        on_hit=True
        var.set("you hit me")
    else:
        on_hit=False
        var.set("")

#添加button,conmmand指定点击后需要运行的函数
b=tk.Button(windows,text="hit me",width=15,height=2,command=hit_me)
#安置button
b.pack()

windows.mainloop()

2.Entry &Text

import tkinter as tk
windows=tk.TK()
windows.title()
windows=geometry('num x num ')

#定义一个entry
e=tk.Entry(windows,show="*")
e.pack()

#函数
def insert_point():
    var=e.get() #固定方法
    t.insert("insert",var)
    
def insert_end():
    var=e.get()
    t.insert(1.1,var)#在第一行第一列输入

#添加button,conmmand指定点击后需要运行的函数
b1=tk.Button(windows,text="insert point",width=15,height=2,command=insert_point)
#安置button
b1.pack()

b2=tk.Button(windows,text="insert end",width=15,height=2,command=insert_end)
b2.pack()

#添加TEXT
t=tk.Text(windows,height=2)
t.pack()

windows.mainloop()

3. Listbox列表

import tkinter as tk

windows=tk.TK()
windows.title()
windows=geometry('num x num ')

var1=tk.StringVar()
l=tk.Label(windows,bg="yellow",width=4,textvariable=var1)
l.pack()

def print_selection():
    value=lb.get(lb.curselection()) #拿出光标选定的东西
    var1.set(value)

b1=tk.Button(windows,text="print selection",width=15,height=2,command=print_selection)
#安置button
b1.pack()

var2=tk.StringVar()
var2.set((11,22,33,44))
#定义列表部件，第一种插入方法
lb=tk.Listbox(windows,listvariable=var2)
#第二种插入方法
list_items=[1,2,3,4]
for item in list_items:
    lb.insert("end",item)
#第三种插入法
lb.insert(1,"first")
lb.delete(2)
lb.pack()

windows.mainloop()

4.Radiobutton选择按钮

import tkinter as tk

windows=tk.TK()
windows.title()
windows=geometry('num x num ')

var=tk.StringVar()
l=tk.Label(windows,bg="yellow",width=4,textvariable=var,text="empty")
l.pack()
def print_selection():
    l.config(text="you have select"+var.get())
#创建radiobutton，如果选中，value=A传给var ,并在label中显示A
r1=tk.Radiobutton(windows,text="Option A",variable=var,value="A",command=print_selection)
r1.pack()

r2=tk.Radiobutton(windows,text="Option B",variable=var,value="B",command=print_selection)
r2.pack()

windows.mainloop()

5.Scale

#可以被拉动的进度条
import tkinter as tk

windows=tk.TK()
windows.title()
windows=geometry('num x num ')

l=tk.Label(windows,bg="yellow",width=4,textvariable=var,text="empty")
l.pack()
def print_selection(v):
    l.config(text="you have select"+v)
#定义scale,label为名字，showvalue为是否显示选定值，
#resolution为小数位 tickinterval为隔多少显示一个
s=tk.Scale(windows,label='try me',from_=5,to=11,orient=tk.HORIZONTAL,
         length=200,showvalue=0,tickinterval=3,resolution=0.01,command=print_selection)
s.pack()

windows.mainloop()

7.canvas画布

#与radiobutton不同，checkbutton可以进行多值选择
import tkinter as tk

windows=tk.TK()
windows.title()
windows=geometry('num x num ')

canvas=tk.Canvas(windows,bg="blue",height=100,wight=200)
#到pack之间可以加入图片
image_file=tk.PhotoImage(file="ins.gif")
#anchor:确定图片的哪个部分定点0.0，
image=canvas.create_image(0,0,anchor="nw",image=image_file)

x0,y0,x1,y1=50,50,80,80
line=canvas.create_line(x0,y0,x1,y1) #画线
oval=canvas.create_oval(x0,y0,x1,y1,fill="red") #画个圆
arc=canvas.create_arc(x0+30,y0+30,x1+30,y1+30,start=0,extent=180)#画个扇形
rect=canvas.create_rectangle(100,30,100+20，30+20)#画个正方行
canvas.pack()

def moveit():
    canvas.move(rect,0,2)#调用move，移动画布元素
    
b=tk.Button(windows,text="move",command=moveit).pack()

windows.mainloop()

11.pack grid place放置位置

import tkinter as tk

windows=tk.TK()
windows.title()
windows=geometry('num x num ')
#side:top button left,right
tk.Label(windows,text=1).pack(side="")

#grid：方格形式
#padxy:内部方位
for i in range(4):
    for j in range(3):
        tk.Label(windows,text=1).grid(row=i,column=j,padx=10,pady=10)
        
#place:指定放置在哪个点上
tk.Label(windows,text=1).place(x=,y=,anchor="nw")
windows.mainloop()
